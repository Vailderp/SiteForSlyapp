3. Определение модели нейросети.

Теперь нужно создать нейросеть.

Мы выберем самую простую нейросеть из двух возможных.

Она называется последовательной.

Для активации скрытых слоёв нейронной сети используем "relu".

Всего входных картинок 30000, размер 7680×4320.


def dog_cat_model():
model = Sequential()
model.add(Conv2D(32, (3, 3), input_shape=(7680, 4320, 3), activation='relu'))
model.add(MaxPooling2D(pool_size=(2, 2)))
model.add(Conv2D(32, (3, 3),input_shape=(7680, 4320, 3), activation='relu'))
model.add(MaxPooling2D(pool_size=(2, 2)))
model.add(Flatten())
model.add(Dense(units=128, activation='relu'))
model.add(Dense(units=1, activation='sigmoid'))
model.compile(optimizer=Adam(),
loss='binary_crossentropy',
metrics=['accuracy'])
return model
